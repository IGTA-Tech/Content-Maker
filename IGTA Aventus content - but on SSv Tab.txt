function generateDailyContent() {
  const startTime = new Date();
  const executionId = Utilities.getUuid();
  
  Logger.log('═══════════════════════════════════════════════════════');
  Logger.log(`🚀 EXECUTION START: ${executionId}`);
  Logger.log(`⏰ Time: ${startTime}`);
  Logger.log('═══════════════════════════════════════════════════════');
  
  try {
    const config = getConfig();
    validateConfiguration(config);
    
    // PHASE 1: Data Collection
    Logger.log('\n📊 PHASE 1: DATA COLLECTION');
    const websiteData = comprehensiveWebsiteScrape(config);
    const exampleContent = fetchExampleContent(config);
    const formerClients = getFormerClientsFromSheet();
    
    // PHASE 2: Current Events Intelligence
    Logger.log('\n📰 PHASE 2: CURRENT EVENTS INTELLIGENCE');
    const currentEvents = intelligentEventGathering(config, formerClients);
    
    // PHASE 3: Content Generation with Retry Logic
    Logger.log('\n✍️ PHASE 3: CONTENT GENERATION');
    let articles = null;
    let attempt = 0;
    
    while (attempt < config.MAX_RETRIES && !articles) {
      attempt++;
      Logger.log(`  📝 Generation attempt ${attempt}/${config.MAX_RETRIES}`);
      
      try {
        articles = generateArticlesWithEnhancedPrompt(
          websiteData, 
          currentEvents, 
          exampleContent,
          config
        );
        
        // PHASE 4: Triple Verification
        Logger.log('\n✅ PHASE 4: TRIPLE VERIFICATION');
        articles = tripleLayerVerification(articles, websiteData, config);
        
      } catch (verificationError) {
        Logger.log(`  ❌ Attempt ${attempt} failed: ${verificationError.message}`);
        
        if (attempt < config.MAX_RETRIES) {
          Logger.log(`  ⏳ Waiting ${config.RETRY_DELAY_MS}ms before retry...`);
          Utilities.sleep(config.RETRY_DELAY_MS);
        } else {
          throw new Error(`Failed after ${config.MAX_RETRIES} attempts: ${verificationError.message}`);
        }
      }
    }
    
    // PHASE 5: Thumbnail Generation (Context-Aware)
    Logger.log('\n🎨 PHASE 5: CONTEXT-AWARE THUMBNAIL GENERATION');
    const thumbnails = generateContextAwareThumbnails(articles, config);
    
    // PHASE 6: Email Delivery
    Logger.log('\n📧 PHASE 6: EMAIL DELIVERY');
    sendEnhancedContentEmail(articles, thumbnails, config, executionId);
    
    // PHASE 7: Archiving & Analytics
    Logger.log('\n💾 PHASE 7: ARCHIVING');
    archiveContentWithMetadata(articles, websiteData, currentEvents);
    updateVerifiedNamesUsage(articles);
    
    const endTime = new Date();
    const duration = (endTime - startTime) / 1000;
    
    Logger.log('\n═══════════════════════════════════════════════════════');
    Logger.log(`✅ EXECUTION COMPLETE: ${executionId}`);
    Logger.log(`⏱️ Total Duration: ${duration} seconds`);
    Logger.log(`📊 Articles Generated: 2`);
    Logger.log(`🎯 Confidence: ${articles.article1.confidence}%, ${articles.article2.confidence}%`);
    Logger.log('═══════════════════════════════════════════════════════');
    
    return {
      success: true,
      executionId: executionId,
      duration: duration,
      articles: articles
    };
    
  } catch (error) {
    handleComprehensiveError(error, executionId, startTime);
    throw error;
  }
}

function validateConfiguration(config) {
  Logger.log('  🔍 Validating configuration...');
  
  const requiredKeys = ['ANTHROPIC_API_KEY', 'OPENAI_API_KEY'];
  const missing = [];
  
  requiredKeys.forEach(key => {
    if (!config[key]) {
      missing.push(key);
    }
  });
  
  if (missing.length > 0) {
    throw new Error(`Missing required Script Properties: ${missing.join(', ')}. Add them in Project Settings → Script Properties`);
  }
  
  // Validate API key formats
  if (!config.ANTHROPIC_API_KEY.startsWith('sk-ant-')) {
    throw new Error('Invalid ANTHROPIC_API_KEY format. Should start with "sk-ant-"');
  }
  
  if (!config.OPENAI_API_KEY.startsWith('sk-')) {
    throw new Error('Invalid OPENAI_API_KEY format. Should start with "sk-"');
  }
  
  Logger.log('  ✅ Configuration valid');
}

// COMPREHENSIVE WEBSITE SCRAPING
// ═══════════════════════════════════════════════════════════════════════

function comprehensiveWebsiteScrape(config) {
  Logger.log('  🌐 Starting comprehensive website scrape...');
  
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('Website_URLs');
  const urls = sheet.getRange(2, 1, sheet.getLastRow() - 1, 1).getValues().flat().filter(url => url);
  
  const scrapedData = {
    cases: [],
    names: new Set(),
    visaTypes: new Set(),
    achievements: [],
    quotes: [],
    dates: [],
    fullContent: [],
    metadata: {
      scrapedAt: new Date(),
      urlsScraped: urls.length,
      successCount: 0,
      failureCount: 0
    }
  };
  
  Logger.log(`  📄 Scraping ${urls.length} URLs...`);
  
  urls.forEach((url, index) => {
    try {
      Logger.log(`    [${index + 1}/${urls.length}] ${url.substring(0, 50)}...`);
      
      const response = UrlFetchApp.fetch(url, {
        muteHttpExceptions: true,
        followRedirects: true
      });
      
      if (response.getResponseCode() !== 200) {
        Logger.log(`    ⚠️ HTTP ${response.getResponseCode()}`);
        scrapedData.metadata.failureCount++;
        return;
      }
      
      const html = response.getContentText();
      
      // Extract names (proper nouns with first and last name)
      const namePattern = /\b([A-Z][a-z]+)\s+([A-Z][a-z]+)\b/g;
      let nameMatch;
      while ((nameMatch = namePattern.exec(html)) !== null) {
        const fullName = nameMatch[0];
        // Filter out common false positives
        if (!fullName.match(/^(United States|Google Drive|New York|Los Angeles)/)) {
          scrapedData.names.add(fullName);
        }
      }
      
      // Extract visa types with context
      const visaPattern = /(P-1|O-1|O-1A|O-1B|EB-1A|H-1B)\s+visa/gi;
      let visaMatch;
      while ((visaMatch = visaPattern.exec(html)) !== null) {
        scrapedData.visaTypes.add(visaMatch[1].toUpperCase());
      }
      
      // Extract quoted text (potential testimonials)
      const quotePattern = /"([^"]{50,300})"/g;
      let quoteMatch;
      while ((quoteMatch = quotePattern.exec(html)) !== null) {
        scrapedData.quotes.push(quoteMatch[1]);
      }
      
      // Extract years (for dating cases)
      const yearPattern = /\b(20\d{2})\b/g;
      let yearMatch;
      while ((yearMatch = yearPattern.exec(html)) !== null) {
        scrapedData.dates.push(yearMatch[1]);
      }
      
      // Store case information
      scrapedData.cases.push({
        url: url,
        title: extractTitle(html),
        content: html.substring(0, 15000), // First 15k chars
        namesFound: [...new Set([...html.matchAll(namePattern)].map(m => m[0]))],
        visasFound: [...new Set([...html.matchAll(visaPattern)].map(m => m[1].toUpperCase()))]
      });
      
      // Store limited full content for Claude analysis
      scrapedData.fullContent.push({
        url: url,
        snippet: extractMainContent(html)
      });
      
      scrapedData.metadata.successCount++;
      
      // Rate limiting
      Utilities.sleep(config.SCRAPING_DELAY_MS);
      
    } catch (e) {
      Logger.log(`    ❌ Error: ${e.message}`);
      scrapedData.metadata.failureCount++;
    }
  });
  
  // Convert Sets to Arrays for JSON serialization
  scrapedData.names = [...scrapedData.names];
  scrapedData.visaTypes = [...scrapedData.visaTypes];
  
  Logger.log(`  ✅ Scraping complete:`);
  Logger.log(`    📊 Success: ${scrapedData.metadata.successCount}`);
  Logger.log(`    ❌ Failures: ${scrapedData.metadata.failureCount}`);
  Logger.log(`    👤 Names found: ${scrapedData.names.length}`);
  Logger.log(`    📋 Visa types: ${scrapedData.visaTypes.length}`);
  
  // Update sheet with last scraped time
  updateLastScrapedTimes(sheet, urls);
  
  return scrapedData;
}

function extractTitle(html) {
  const titleMatch = html.match(/<title>([^<]+)<\/title>/i);
  return titleMatch ? titleMatch[1].trim() : 'Unknown';
}

function extractMainContent(html) {
  // Remove scripts, styles, and nav elements
  let cleaned = html
    .replace(/<script\b[^<]*(?:(?!<\/script>)<[^<]*)*<\/script>/gi, '')
    .replace(/<style\b[^<]*(?:(?!<\/style>)<[^<]*)*<\/style>/gi, '')
    .replace(/<nav\b[^<]*(?:(?!<\/nav>)<[^<]*)*<\/nav>/gi, '')
    .replace(/<header\b[^<]*(?:(?!<\/header>)<[^<]*)*<\/header>/gi, '')
    .replace(/<footer\b[^<]*(?:(?!<\/footer>)<[^<]*)*<\/footer>/gi, '');
  
  // Remove HTML tags
  cleaned = cleaned.replace(/<[^>]+>/g, ' ');
  
  // Clean up whitespace
  cleaned = cleaned.replace(/\s+/g, ' ').trim();
  
  // Return first 5000 chars
  return cleaned.substring(0, 5000);
}

function updateLastScrapedTimes(sheet, urls) {
  const now = Utilities.formatDate(new Date(), 'America/New_York', 'yyyy-MM-dd HH:mm:ss');
  
  for (let i = 0; i < urls.length; i++) {
    sheet.getRange(i + 2, 3).setValue(now);
  }
}

    { type: 'week', weight: 0.5, query: 'athlete visa immigration updates this week' },
    { type: 'month', weight: 0.3, query: 'USCIS policy changes ' + config.CURRENT_YEAR }
  ];
  
  const searchPrompt = `You are researching current sports immigration news for ${config.TODAY}.

SEARCH REQUIREMENTS:
1. Find news from multiple timeframes:
   - TODAY (20%): Breaking news from today
   - THIS WEEK (50%): Recent developments from past 7 days
   - THIS MONTH (30%): Significant changes from past 30 days

2. Focus areas:
   - USCIS policy changes affecting athletes
   - P-1, O-1, EB-1A visa updates
   - Major sports immigration stories
   - Athletes having visa issues or successes

3. Former client monitoring:
   Check if any of these clients have recent news:
   ${formerClients.map(name => `   - ${name}`).join('\n')}

4. Event matching criteria:
   - Relevance to sports visas
   - Connection to SSV's expertise
   - Potential to pair with historical cases
   - Timeliness and newsworthiness

RETURN FORMAT:
Provide a structured summary with:
- Event headline and date
- Relevance score (1-10)
- Connection to visa types (P-1, O-1, EB-1A)
- Potential historical case matches
- Former client updates (if any)

Be specific with dates and sources. Focus on quality over quantity.`;

  try {
    const response = UrlFetchApp.fetch('https://api.anthropic.com/v1/messages', {
      method: 'post',
      headers: {
        'x-api-key': config.ANTHROPIC_API_KEY,
        'anthropic-version': '2023-06-01',
        'content-type': 'application/json'
      },
      payload: JSON.stringify({
        model: 'claude-sonnet-4-20250514',
        max_tokens: 3000,
        temperature: 0.7,
        messages: [{
          role: 'user',
          content: searchPrompt
        }]
      }),
      muteHttpExceptions: true
    });
    
    const data = JSON.parse(response.getContentText());
    
    if (!data.content || !data.content[0]) {
      Logger.log('  ⚠️ No current events found, using fallback mode');
      return createFallbackEvents(config, formerClients);
    }
    
    const events = {
      raw: data.content[0].text,
      timeframes: timeframes,
      formerClientsChecked: formerClients,
      searchDate: config.TODAY

    Logger.log('  ✅ Current events gathered successfully');

    return events;

    Logger.log(`  ❌ Error gathering events: ${e.message}`);
    return createFallbackEvents(config, formerClients);
  }
}

    timeframes: [{ type: 'general', weight: 1.0 }],
    formerClientsChecked: formerClients,
    searchDate: config.TODAY,
    fallbackMode: true
  };
}

      }),
      muteHttpExceptions: true
    });
    
    const data = JSON.parse(response.getContentText());
    
    if (!data.content || !data.content[0]) {
      throw new Error('Claude API returned empty response: ' + JSON.stringify(data));
    }
    
    // Extract and parse JSON
    let jsonText = data.content[0].text;
    
    // Remove markdown code blocks if present
    jsonText = jsonText.replace(/```json\s*/g, '').replace(/```\s*/g, '').trim();
    
    // Find JSON object
    const jsonStart = jsonText.indexOf('{');
    const jsonEnd = jsonText.lastIndexOf('}');
    
    if (jsonStart === -1 || jsonEnd === -1) {
      throw new Error('No JSON object found in Claude response');
    }
    
    jsonText = jsonText.substring(jsonStart, jsonEnd + 1);
    
    const articles = JSON.parse(jsonText);
    
    // Validate structure
    if (!articles.article1 || !articles.article2) {
      throw new Error('Invalid article structure in response');
    }
    
    Logger.log('  ✅ Articles generated successfully');
    Logger.log(`    📝 Article 1: "${articles.article1.title}" (${articles.article1.wordCount} words, ${articles.article1.confidence}% confidence)`);
    Logger.log(`    📝 Article 2: "${articles.article2.title}" (${articles.article2.wordCount} words, ${articles.article2.confidence}% confidence)`);
    
    return articles;
    
  } catch (e) {
    Logger.log(`  ❌ Generation error: ${e.message}`);
    throw new Error(`Content generation failed: ${e.message}`);
  }
}

// TRIPLE-LAYER VERIFICATION SYSTEM
// ═══════════════════════════════════════════════════════════════════════

function tripleLayerVerification(articles, websiteData, config) {
  Logger.log('  🔍 Starting triple-layer verification...');
  
  // TIER 1: Confidence Score Check
  Logger.log('    [1/3] Checking confidence scores...');
  if (articles.article1.confidence < config.CONFIDENCE_THRESHOLD) {
    throw new Error(`Article 1 confidence too low: ${articles.article1.confidence}% (threshold: ${config.CONFIDENCE_THRESHOLD}%)`);
  }
  
  if (articles.article2.confidence < config.CONFIDENCE_THRESHOLD) {
    throw new Error(`Article 2 confidence too low: ${articles.article2.confidence}% (threshold: ${config.CONFIDENCE_THRESHOLD}%)`);
  }
  
  Logger.log(`    ✓ Confidence scores pass (${articles.article1.confidence}%, ${articles.article2.confidence}%)`);
  
  // TIER 2: Name Verification
  Logger.log('    [2/3] Verifying all names...');
  const approvedNames = getVerifiedNames();
  const allScrapedNames = [...websiteData.names, ...approvedNames];
  
  const verifyArticleNames = (article, articleNum) => {
    article.namesUsed.forEach(name => {
      if (!allScrapedNames.includes(name)) {
        throw new Error(`Unapproved name in Article ${articleNum}: "${name}". This name was not found in website scraping or verified names list.`);
      }
    });
  };
  
  verifyArticleNames(articles.article1, 1);
  verifyArticleNames(articles.article2, 2);
  
  Logger.log(`    ✓ All names verified (${articles.article1.namesUsed.length + articles.article2.namesUsed.length} total)`);
  
  // TIER 3: Attorney Name & Critical Elements
  Logger.log('    [3/3] Verifying critical elements...');
  
  const checkCriticalElements = (article, articleNum) => {
    const content = article.content.toLowerCase();
    
    // Check attorney name spelling
    if (content.includes('sherrod') && !content.includes('sherrod seward')) {
      throw new Error(`Article ${articleNum}: Attorney name misspelled. Must be "Sherrod Seward"`);
    }
    
    // Check word count
    if (article.wordCount < config.MIN_WORD_COUNT) {
      throw new Error(`Article ${articleNum}: Word count too low (${article.wordCount} < ${config.MIN_WORD_COUNT})`);
    }
    
    if (article.wordCount > config.MAX_WORD_COUNT) {
      throw new Error(`Article ${articleNum}: Word count too high (${article.wordCount} > ${config.MAX_WORD_COUNT})`);
    }
    
    // Check required elements
    if (!article.title || article.title.length > 60) {
      throw new Error(`Article ${articleNum}: Invalid title (must be under 60 chars)`);
    }
    
    if (!article.metaDescription || article.metaDescription.length !== 155) {
      throw new Error(`Article ${articleNum}: Meta description must be exactly 155 characters (is ${article.metaDescription.length})`);
    }
    
    if (!article.content.includes('# ')) {
      throw new Error(`Article ${articleNum}: Missing H1 heading`);
    }
    
    if (!article.content.includes('## ')) {
      throw new Error(`Article ${articleNum}: Missing H2 headings`);
    }
  };
  
  checkCriticalElements(articles.article1, 1);
  checkCriticalElements(articles.article2, 2);
  
  Logger.log('    ✓ Critical elements verified');
  Logger.log('  ✅ Triple-layer verification complete - All checks passed');
  
  return articles;
}

  return sheet.getRange(2, 1, sheet.getLastRow() - 1, 1).getValues().flat();
}

function setupDailyTrigger() {
  Logger.log('Setting up daily trigger...');
  
  // Delete all existing triggers for this project
  const triggers = ScriptApp.getProjectTriggers();
  triggers.forEach(trigger => {
    Logger.log(`Deleting existing trigger: ${trigger.getHandlerFunction()}`);
    ScriptApp.deleteTrigger(trigger);
  });
  
  // Create new 4 AM EST trigger
  ScriptApp.newTrigger('generateDailyContent')
    .timeBased()
    .atHour(4)
    .everyDays(1)
    .inTimezone('America/New_York')
    .create();
  
  Logger.log('✅ Daily trigger created for 4:00 AM EST');
  SpreadsheetApp.getUi().alert(
    '✅ Trigger Set Successfully!\n\n' +
    'Daily content generation will run at 4:00 AM EST.\n\n' +
    'You will receive an email with 2 articles every morning.'
  );
}

function deleteDailyTrigger() {
  const triggers = ScriptApp.getProjectTriggers();
  let deleted = 0;
  
  triggers.forEach(trigger => {
    ScriptApp.deleteTrigger(trigger);
    deleted++;
  });
  
  SpreadsheetApp.getUi().alert(`Deleted ${deleted} trigger(s)`);
}

function viewLastError() {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('Error_Log');
  const lastRow = sheet.getLastRow();
  
  if (lastRow < 2) {
    SpreadsheetApp.getUi().alert('✅ No Errors Logged\n\nYour system is running cleanly!');
    return;
  }
  
  const error = sheet.getRange(lastRow, 1, 1, 5).getValues()[0];
  
  const report = `

Time: ${error[0]}
Type: ${error[1]}
Stage: ${error[2]}
Message: ${error[3]}

Check Apps Script logs for full details:
Extensions → Apps Script → View → Executions
